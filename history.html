import { auth, db } from "./firebase.js"; import { onAuthStateChanged } from "https://www.gstatic.com/firebasejs/12.1.0/firebase-auth.js"; import { collection, query, where, getDocs, orderBy } from "https://www.gstatic.com/firebasejs/12.1.0/firebase-firestore.js"; const depositHistory = document.getElementById("depositHistory"); const withdrawHistory = document.getElementById("withdrawHistory"); const investmentHistory = document.getElementById("investmentHistory"); // ============================== // CHECK LOGIN // ============================== onAuthStateChanged(auth, async (user)=>{ if(!user){ window.location.href = "login.html"; return; } try{ await loadDeposits(user.uid); await loadWithdrawals(user.uid); await loadInvestments(user.uid); } catch(error){ console.error(error); alert(error.message); } }); // ============================== // LOAD DEPOSITS // ============================== async function loadDeposits(uid){ const q = query( collection(db,"depositRequests"), where("userId","==",uid) ); const snapshot = await getDocs(q); if(snapshot.empty){ depositHistory.innerHTML = "

No deposit history found.

"; return; } depositHistory.innerHTML = ""; snapshot.forEach((doc)=>{ const data = doc.data(); depositHistory.innerHTML += `

Deposit

Amount: ₦${Number(data.amount).toLocaleString()}

Method: ${data.method}

Reference: ${data.reference}

Status: ${statusBadge(data.status)}

Date: ${formatDate(data.createdAt)}

`; }); } // ============================== // LOAD WITHDRAWALS // ============================== async function loadWithdrawals(uid){ const q = query( collection(db,"withdrawRequests"), where("userId","==",uid) ); const snapshot = await getDocs(q); if(snapshot.empty){ withdrawHistory.innerHTML = "

No withdrawal history found.

"; return; } withdrawHistory.innerHTML = ""; snapshot.forEach((doc)=>{ const data = doc.data(); withdrawHistory.innerHTML += `

Withdrawal

Amount: ₦${Number(data.amount).toLocaleString()}

Bank: ${data.bank}

Account: ${data.accountNumber}

Status: ${statusBadge(data.status)}

Date: ${formatDate(data.createdAt)}

`; }); } // ============================== // LOAD INVESTMENTS // ============================== async function loadInvestments(uid){ const q = query( collection(db,"investments"), where("userId","==",uid) ); const snapshot = await getDocs(q); if(snapshot.empty){ investmentHistory.innerHTML = "

No investment history found.

"; return; } investmentHistory.innerHTML = ""; snapshot.forEach((doc)=>{ const data = doc.data(); investmentHistory.innerHTML += `

${data.planName}

Investment: ₦${Number(data.investmentAmount).toLocaleString()}

Daily Profit: ₦${Number(data.dailyProfit).toLocaleString()}

Total Profit: ₦${Number(data.totalProfit || 0).toLocaleString()}

Status: ${statusBadge(data.status)}

Date: ${formatDate(data.createdAt)}

`; }); } // ============================== // STATUS // ============================== function statusBadge(status){ if(status === "Approved" || status === "Active"){ return ` 🟢 ${status} `; } if(status === "Rejected"){ return ` 🔴 Rejected `; } return ` 🟡 Pending `; } // ============================== // DATE // ============================== function formatDate(date){ if(!date){ return "-"; } return new Date(date).toLocaleString(); }
